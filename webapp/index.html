Automated PowerShell Cleanup Runbook (AI-Generated):
# Azure CloudOps AI - Automated Cleanup Runbook
# Targets unattached Managed Disks and Public IPs
Write-Host "Connecting to Azure Subscription..."

# Fetch orphaned Managed Disks
$orphanedDisks = Get-AzDisk | Where-Object { $_.DiskState -eq 'Unattached' }
foreach ($disk in $orphanedDisks) {
    Write-Host "Removing unattached disk: $($disk.Name) (Saving $120.00/mo)" -ForegroundColor Yellow
    Remove-AzDisk -ResourceGroupName $disk.ResourceGroupName -DiskName $disk.Name -Force -ErrorAction SilentlyContinue
}

# Fetch unassociated Public IP Addresses
$ips = Get-AzPublicIpAddress | Where-Object { $_.IpConfiguration -eq $null }
foreach ($ip in $ips) {
    Write-Host "Removing unused Public IP: $($ip.Name) (Saving $5.00/mo)" -ForegroundColor Yellow
    Remove-AzPublicIpAddress -ResourceGroupName $ip.ResourceGroupName -Name $ip.Name -Force -ErrorAction SilentlyContinue
}

Write-Host "Cleanup completed successfully!" -ForegroundColor Green
